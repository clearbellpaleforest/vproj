# Vim Project Navigator (VPN) — Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Phase 1 MVP of VPN — sidebar opens via `<F2>`, shows buffers with single-keystroke labels, mode switching works, and pressing a label key jumps to that buffer immediately.

**Architecture:** Neovim >=0.10 Lua plugin. Floating window sidebar with virtual buffer rendering. Event bus decouples subsystems. Label engine generates 36 direct-select keys mapped to items. Mode registry allows pluggable views. Buffer mode is the first mode implemented.

**Tech Stack:** Lua (Neovim native), plenary.nvim (testing), Neovim API (nvim_open_win, nvim_buf_set_lines, nvim_buf_set_extmark, vim.api)

```lua
    config = require("vpn.config").setup(user_opts)
    events = require("vpn.utils.events")

    else
        M.open()
    end

    sidebar.open()
    local first_mode = mode_registry.get_default()
    mode_registry.switch(first_mode.key)
end

function M.close()
    sidebar.close()
end

return M
```

```lua
-- lua/vpn/utils/events.lua
local M = {}

local listeners = {}
local next_id = 0

function M.on(event, fn)
    if not listeners[event] then
        listeners[event] = {}
    end
    next_id = next_id + 1
    local id = next_id
    listeners[event][id] = fn
    return id
end

function M.off(event, id)
    if listeners[event] then
        listeners[event][id] = nil
    end
end

function M.emit(event, data)
    if listeners[event] then
        for _, fn in pairs(listeners[event]) do
            fn(data)
        end
    end
end

function M.clear()
    listeners = {}
    next_id = 0
end

return M
```

```lua
    hotkey = "<F2>",

    labels = {

            "1234567890",
            "asdfghjkl",

        symbols = { enabled = true },

    },

        auto_restore = true,
    },
    cache = {
        project_ttl = 30,
        git_ttl = 5,
        lsp_ttl = 10,
    },
}

local function deep_merge(base, override)
    local result = vim.deepcopy(base)
    for k, v in pairs(override or {}) do
        if type(v) == "table" and type(result[k]) == "table" then
            result[k] = deep_merge(result[k], v)
        else
            result[k] = v
        end
    end
    return result
end

function M.setup(user_opts)
    local cfg = deep_merge(defaults, user_opts)
    return cfg
end

return M
```

```lua
-- lua/vpn/utils/cache.lua
local M = {}

function M.new(opts)
    local ttl = (opts and opts.ttl) or 60
    local store = {}

    local self = {}

    function self.set(key, value)
        store[key] = {
            value = value,
            expires_at = os.time() + ttl,
        }
    end

    function self.get(key)
        local entry = store[key]
        if not entry then
            return nil
        end
        if os.time() > entry.expires_at then
            store[key] = nil
            return nil
        end
        return entry.value
    end

    function self.invalidate(key)
        store[key] = nil
    end

    function self.clear()
        store = {}
    end

    return self
end

return M
```

```lua
-- lua/vpn/ui/labels.lua
local M = {}

function M.generate(count, tiers, overflow_style)
    local labels = {}
    local idx = 1
    local flat_chars = {}

    for _, tier in ipairs(tiers) do
        for ch in tier:gmatch(".") do
            flat_chars[#flat_chars + 1] = ch
        end
    end

    for _, ch in ipairs(flat_chars) do
        if idx > count then break end
        labels[idx] = ch
        idx = idx + 1
    end

    if idx <= count and overflow_style == "double" then
        for _, c1 in ipairs(flat_chars) do
            for _, c2 in ipairs(flat_chars) do
                if idx > count then break end
                labels[idx] = c1 .. c2
                idx = idx + 1
            end
            if idx > count then break end
        end
    end

    return labels
end

function M.build_map(items, tiers, overflow_style)
    local count = #items
    local lbls = M.generate(count, tiers, overflow_style)
    local label_map = {}
    local lines = {}

    for i, item in ipairs(items) do
        local label = lbls[i]
        label_map[label] = item
        lines[i] = string.format("%s %s", label, item.name)
    end

    return {
        label_map = label_map,
        lines = lines,
    }
end

return M
```

```lua
-- lua/vpn/ui/sidebar.lua
local M = {}

local win_id = nil
local buf_id = nil
local config = nil

function M.setup(cfg)
    config = cfg
end

function M.is_open()
    return win_id ~= nil and vim.api.nvim_win_is_valid(win_id)
end

function M.get_buf()
    if buf_id and vim.api.nvim_buf_is_valid(buf_id) then
        return buf_id
    end
    return nil
end

function M.open()
    if M.is_open() then
        return
    end

    buf_id = vim.api.nvim_create_buf(false, true)
    vim.bo[buf_id].buftype = "nofile"
    vim.bo[buf_id].bufhidden = "wipe"
    vim.bo[buf_id].modifiable = true
    vim.bo[buf_id].filetype = "vpn-sidebar"

    local width = config and config.width or 35
    local height = vim.o.lines - 2
    win_id = vim.api.nvim_open_win(buf_id, false, {
        relative = "editor",
        width = width,
        height = height,
        row = 0,
        col = 0,
        style = "minimal",
        border = "single",
    })

    vim.wo[win_id].number = false
    vim.wo[win_id].relativenumber = false
    vim.wo[win_id].cursorline = false
    vim.wo[win_id].signcolumn = "no"
end

function M.close()
    if not M.is_open() then
        return
    end
    vim.api.nvim_win_close(win_id, true)
    win_id = nil
    buf_id = nil
end

function M.get_win()
    return win_id
end

return M
```

```lua
-- lua/vpn/ui/renderer.lua
local M = {}

local ns_id = vim.api.nvim_create_namespace("vpn-renderer")

function M.render(buf_id, header, lines, label_map)
    if not buf_id or not vim.api.nvim_buf_is_valid(buf_id) then
        return
    end

    vim.bo[buf_id].modifiable = true

    local content = {}
    if header then
        vim.list_extend(content, header)
    end
    local separator = string.rep("─", 35)
    table.insert(content, separator)

    vim.list_extend(content, lines)

    vim.api.nvim_buf_set_lines(buf_id, 0, -1, false, content)
    vim.bo[buf_id].modifiable = false
end

function M.render_header(buf_id, mode_name, modes)
    if not buf_id or not vim.api.nvim_buf_is_valid(buf_id) then
        return
    end

    local header = {}
    local parts = {}
    for _, mode in ipairs(modes) do
        if mode.enabled then
            table.insert(parts, string.format("[%s]%s", mode.key, mode.name))
        end
    end
    table.insert(header, "PROJECT [" .. mode_name .. "]")
    table.insert(header, "")
    table.insert(header, table.concat(parts, "  "))

    vim.bo[buf_id].modifiable = true
    vim.api.nvim_buf_set_lines(buf_id, 0, -1, false, header)
    vim.bo[buf_id].modifiable = false
end

function M.render_items(buf_id, lines)
    if not buf_id or not vim.api.nvim_buf_is_valid(buf_id) then
        return
    end

    local offset = vim.api.nvim_buf_line_count(buf_id)
    vim.bo[buf_id].modifiable = true
    vim.api.nvim_buf_set_lines(buf_id, offset, -1, false, lines)
    vim.bo[buf_id].modifiable = false
end

function M.clear(buf_id)
    if not buf_id or not vim.api.nvim_buf_is_valid(buf_id) then
        return
    end
    vim.bo[buf_id].modifiable = true
    vim.api.nvim_buf_set_lines(buf_id, 0, -1, false, {})
    vim.bo[buf_id].modifiable = false
end

return M
```

```lua
local modes = {}
local current = nil
local config = nil
local events = nil

function M.setup(cfg, ev)
    config = cfg
    events = ev
end

function M.register(mode)
    modes[mode.key] = mode
end

function M.get(key)
    return modes[key]
end

function M.switch(key)
    local mode = modes[key]
    if not mode then
        return nil
    end
    current = mode
    if mode.refresh then
        mode.refresh()
    end
    events.emit("mode_changed", { mode = mode })
    return mode
end

function M.get_current()
    return current
end

function M.get_default()
    for _, mode in pairs(modes) do
        if mode.enabled then
            return mode
        end
    end
    return nil
end

function M.all()
    local result = {}
    for _, mode in pairs(modes) do
        table.insert(result, mode)
    end
    return result
end

return M
```

```lua
-- lua/vpn/modes/buffers.lua
local M = {}

local labels = require("vpn.ui.labels")
local tiers = { "1234567890", "asdfghjkl", "qwertyuiop", "zxcvbnm" }

function M.create()
    local self = {
        name = "Buffers",
        key = "b",
        icon = "B",
        enabled = true,
        items = {},
        label_map = {},
        lines = {},
    }

    function self.get_status(buf)
        local status = ""
        if vim.bo[buf].modified then
            status = status .. "*"
        end
        if vim.bo[buf].readonly then
            status = status .. "R"
        end
        if vim.api.nvim_buf_get_option(buf, "buftype") == "terminal" then
            status = status .. "T"
        end
        return status
    end

    function self.refresh()
        self.items = {}
        local buffers = vim.api.nvim_list_bufs()
        for _, buf in ipairs(buffers) do
            if vim.api.nvim_buf_is_valid(buf) and vim.bo[buf].buflisted then
                local name = vim.api.nvim_buf_get_name(buf)
                if name == "" then
                    name = "[No Name]"
                else
                    name = vim.fn.fnamemodify(name, ":t")
                end
                table.insert(self.items, {
                    name = name,
                    buf = buf,
                    status = self.get_status(buf),
                })
            end
        end
    end

    function self.render()
        local result = labels.build_map(self.items, tiers, "double")
        self.label_map = result.label_map
        self.lines = result.lines
        return result
    end

    function self.select(label)
        local item = self.label_map[label]
        if not item then
            return false
        end
        if item.buf and vim.api.nvim_buf_is_valid(item.buf) then
            vim.api.nvim_set_current_buf(item.buf)
            return true
        end
        return false
    end

    return self
end

return M
```

```lua
-- lua/vpn/core/navigation.lua
local M = {}

local config = nil
local events = nil
local handler = nil

function M.setup(cfg, ev)
    config = cfg
    events = ev
end

function M.attach(buf_id)
    if not buf_id or not vim.api.nvim_buf_is_valid(buf_id) then
        return
    end

    vim.bo[buf_id].buftype = "nofile"
    vim.bo[buf_id].modifiable = false

    local all_chars = {}
    for _, tier in ipairs(config.labels.tiers) do
        for ch in tier:gmatch(".") do
            all_chars[#all_chars + 1] = ch
        end
    end

    for _, ch in ipairs(all_chars) do
        vim.keymap.set("n", ch, function()
            M.dispatch(ch)
        end, { buffer = buf_id, nowait = true, silent = true })
    end

    local mode_keys = {}
    for _, mode_key in ipairs({ "b", "f", "s", "g", "o" }) do
        mode_keys[mode_key] = true
    end

    -- Mode switching keys (handled at a higher level)
    vim.keymap.set("n", "q", function()
        require("vpn").close()
    end, { buffer = buf_id, nowait = true, silent = true, desc = "Close sidebar" })

    -- Page navigation keys
    vim.keymap.set("n", "[", function()
        events.emit("page_prev")
    end, { buffer = buf_id, nowait = true, silent = true, desc = "Previous page" })

    vim.keymap.set("n", "]", function()
        events.emit("page_next")
    end, { buffer = buf_id, nowait = true, silent = true, desc = "Next page" })
end

function M.set_handler(fn)
    handler = fn
end

function M.dispatch(label)
    if not handler then
        return false
    end
    return handler(label)
end

return M
```

```lua
-- lua/vpn/init.lua
local M = {}

local config = nil
local sidebar = nil
local navigation = nil
local mode_registry = nil
local events = nil
local renderer = nil

function M.setup(user_opts)
    config = require("vpn.config").setup(user_opts or {})
    events = require("vpn.utils.events")

    sidebar = require("vpn.ui.sidebar")
    sidebar.setup(config)

    renderer = require("vpn.ui.renderer")

    navigation = require("vpn.core.navigation")
    navigation.setup(config, events)

    mode_registry = require("vpn.modes")
    mode_registry.setup(config, events)

    -- Register buffer mode
    local buffers = require("vpn.modes.buffers")
    local buffer_mode = buffers.create()
    mode_registry.register(buffer_mode)

    -- Wire mode switching via mode keys
    events.on("mode_changed", function(data)
        if sidebar.is_open() then
            local buf = sidebar.get_buf()
            if buf then
                renderer.clear(buf)
                renderer.render_header(buf, data.mode.name, mode_registry.all())
                local result = data.mode.render()
                if result then
                    renderer.render_items(buf, result.lines)
                    navigation.set_handler(function(label)
                        local ok = data.mode.select(label)
                        if not ok then
                            -- Try mode switching by key
                            local switch_mode = mode_registry.get(label)
                            if switch_mode then
                                mode_registry.switch(label)
                            end
                        end
                    end)
                end
            end
        end
    end)

    -- Toggle sidebar
    vim.keymap.set("n", config.hotkey, M.toggle, { desc = "Toggle FenProject sidebar" })
end

function M.toggle()
    if sidebar.is_open() then
        M.close()
    else
        M.open()
    end
end

function M.open()
    sidebar.open()
    local buf = sidebar.get_buf()
    if buf then
        navigation.attach(buf)
    end

    -- Default to buffer mode
    local default_mode = mode_registry.get("b")
    if not default_mode then
        default_mode = mode_registry.get_default()
    end
    if default_mode then
        mode_registry.switch(default_mode.key)
    end
end

function M.close()
    sidebar.close()
end

return M
```
